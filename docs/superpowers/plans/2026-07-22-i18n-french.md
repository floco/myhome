# Multi-language Support (i18n) + French — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `svelte-i18n`-based internationalization to `@myhome/editor` and translate every static UI string across all ~95 components into French, with a Settings-page language switcher persisted in `localStorage`.

**Architecture:** `svelte-i18n` registers two lazy-loaded JSON catalogs (`en.json`, `fr.json`) keyed by module namespace. `main.ts` initializes the library and waits for the locale before mounting `App`. A new `locale.ts` (mirroring the existing `theme.ts` pattern) handles `localStorage` persistence and browser-language detection. Components read strings via the `$_` store (`.svelte` files) or `get(_)` (plain `.ts` files). Migration proceeds module-by-module so each task lands a self-contained, independently testable slice.

**Tech Stack:** Svelte 5, Vite, Vitest, `svelte-i18n` (new dependency, v4.x).

## Global Constraints

- Full design spec: `docs/superpowers/specs/2026-07-22-i18n-french-design.md`. Every requirement in that spec is satisfied by the tasks below.
- **Default locale stays `en`**, and every English catalog value must be an **exact character-for-character copy** of the string it replaces. This is required so the ~600 existing frontend tests (which assert on rendered English text) keep passing unchanged once a component is migrated.
- **Catalog file locations:** `packages/editor/src/lib/locales/en.json` and `packages/editor/src/lib/locales/fr.json`. Both files are edited together in every task — never add a key to one without the other.
- **Key naming:** nested JSON objects, namespaced by module: `common.*`, `nav.*`, `settings.<panel>.*`, `auth.*`, `chores.*`, `costs.*`, `inventory.*`, `works.*`, `kb.*`, `properties.*`, `consumables.*`, `locations.*`, `home.*`, `floorPlan.*` (for Canvas-adjacent chrome), `mediaGallery.*`, `emojiPicker.*`, `markdownEditor.*`, `datePicker.*`. svelte-i18n flattens nested JSON into dot-path keys automatically, e.g. `{ "chores": { "emptyState": "..." } }` is read as `$_('chores.emptyState')`.
- **Reuse `common.*` for generic labels** used across modules — `common.save`, `common.cancel`, `common.delete`, `common.edit`, `common.close`, `common.add`, `common.loading`, `common.confirm` — instead of duplicating the same English/French pair per module.
- **The migration procedure for every component-touching task is the same** (established fully in Task 3, referenced by name in every later task):
  1. Open the file. Find every hardcoded user-facing string literal: markup text nodes, `placeholder`/`title`/`aria-label` attributes, `alert()`/`confirm()`/validation-message strings in `<script>`.
  2. For each string, add a key under the task's namespace to **both** `en.json` (exact copy of the current string) and `fr.json` (accurate French translation) — reusing an existing `common.*` key instead of creating a duplicate when the string is generic (Save/Cancel/Delete/Edit/Close/etc.).
  3. In markup/attributes, replace the literal with `{$_('namespace.key')}`. For strings with dynamic parts (e.g. `` `${n}d overdue` ``), use ICU interpolation: catalog value `"{n}d overdue"`, call site `$_('chores.overdueDays', { values: { n } })`.
  4. In a `.svelte` `<script>` block, `import { _ } from "svelte-i18n"` and call `$_('key')` directly (Svelte's `$`-auto-subscription works anywhere in a component, including inside event handlers, in both legacy and runes mode) — no `get()` needed.
  5. In a plain `.ts` module (not a `.svelte` file), `import { _ } from "svelte-i18n"; import { get } from "svelte/store";` and call `get(_)('key')`, since there's no compiler-driven auto-subscription outside `.svelte` files.
  6. Never translate user-entered data (item names, notes, addresses, KB content) — only app-authored chrome text.
- **Test parity check:** after migrating a file, run its existing test file(s) (named in each task) unmodified — they must still pass, proving the English catalog text is byte-for-byte identical to what was hardcoded before.
- **Out of scope** (do not touch): backend-generated strings (`packages/backend`), `lib/countryFlags.ts` (~170 country names), `lib/furnitureLibrary.ts` (~37 furniture labels + 7 categories) — both stay English-only per the approved spec.

- [ ] **Step 4: Write `locale.ts`**

```ts
// packages/editor/src/lib/locale.ts
import { locale as i18nLocale } from "svelte-i18n";

/** Supported UI locales. */
export type Locale = "en" | "fr";

const STORAGE_KEY = "myhome-locale";

/** Reads the persisted locale from localStorage, falling back to browser-language
 *  detection (French if navigator.language starts with "fr", else English). */
export function getStoredLocale(): Locale {
  const stored = localStorage.getItem(STORAGE_KEY);
  if (stored === "en" || stored === "fr") return stored;
  return navigator.language.toLowerCase().startsWith("fr") ? "fr" : "en";
}

/** Persists the locale to localStorage and applies it via svelte-i18n. */
export function setLocale(locale: Locale): void {
  localStorage.setItem(STORAGE_KEY, locale);
  i18nLocale.set(locale);
}
```

```ts
import { describe, it, expect } from "vitest";
import en from "../src/lib/locales/en.json";
import fr from "../src/lib/locales/fr.json";

function flattenKeys(obj: unknown, prefix = ""): string[] {
  if (typeof obj !== "object" || obj === null) return [prefix];
  return Object.entries(obj).flatMap(([k, v]) => flattenKeys(v, prefix ? `${prefix}.${k}` : k));
}

describe("i18n catalog completeness", () => {
  it("en.json and fr.json have exactly the same set of keys", () => {
    const enKeys = new Set(flattenKeys(en));
    const frKeys = new Set(flattenKeys(fr));

    const missingInFr = [...enKeys].filter((k) => !frKeys.has(k));
    const missingInEn = [...frKeys].filter((k) => !enKeys.has(k));

    expect(missingInFr, `keys missing in fr.json: ${missingInFr.join(", ")}`).toEqual([]);
    expect(missingInEn, `keys missing in en.json: ${missingInEn.join(", ")}`).toEqual([]);
  });

  it("no catalog value is an empty string", () => {
    const emptyEn = flattenKeys(en).filter((k) => {
      const v = k.split(".").reduce((o: any, p) => o?.[p], en);
      return typeof v === "string" && v.trim() === "";
    });
    const emptyFr = flattenKeys(fr).filter((k) => {
      const v = k.split(".").reduce((o: any, p) => o?.[p], fr);
      return typeof v === "string" && v.trim() === "";
    });
    expect(emptyEn).toEqual([]);
    expect(emptyFr).toEqual([]);
  });
});
```

```ts
// packages/editor/src/lib/choreFormat.ts
import { _ } from "svelte-i18n";
import { get } from "svelte/store";
import type { Chore } from "./choreStore.svelte";

export function displayName(chore: Chore): string {
  let name = chore.name.trim();
  if (chore.emoji && name.startsWith(chore.emoji)) name = name.slice(chore.emoji.length).trim();
  return name;
}

export function formatDue(iso: string): string {
  if (!iso) return "—";
  const d = new Date(iso);
  const now = new Date();
  const diffDays = Math.round((d.getTime() - now.getTime()) / 86400000);
  const t = get(_);
  if (diffDays < -1) return t("chores.dueLabel.overdue", { values: { n: Math.abs(diffDays) } });
  if (diffDays === -1) return t("chores.dueLabel.yesterday");
  if (diffDays === 0) return t("chores.dueLabel.today");
  if (diffDays === 1) return t("chores.dueLabel.tomorrow");
  if (diffDays <= 7) return t("chores.dueLabel.inDays", { values: { n: diffDays } });
  return d.toLocaleDateString(undefined, { month: "short", day: "numeric" });
}
```

```ts
import { mount, unmount, flushSync } from "svelte";

import ChoreRow from "../src/lib/components/ChoreRow.svelte";

describe("French locale smoke test", () => {

    await waitLocale();

    unmount(comp);

  });
});
```
